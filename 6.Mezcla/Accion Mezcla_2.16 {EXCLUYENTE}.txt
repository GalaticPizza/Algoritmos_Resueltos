		SINO

		 		LEER(ARCH_PEL, REG_PEL);
		 	SINO
		 		SI REG_PEL.NRO_P = REG_NUE.NRO_P 	ENTONCES
		 			REG_SAL:= REG_PEL;
		 			ESCRIBIR(ARCH_SAL,REG_SAL);
		 			LEER(ARCH_NUE,REG_NUE);
		 			LEER(ARCH_PEL,REG_PEL);
		 		FS;
		 	FS;
		FS;
	FM;
	
	MIENTRAS NFDA(ARCH_NUE) HACER
		REG_SAL:= REG_NUE;
		ESCRIBIR(ARCH_SAL,REG_SAL);
		LEER(ARCH_NUE, REG_NUE);
	FM;

	MIENTRAS NFDA(ARCH_PEL) HACER
		REG_SAL:= REG_PEL;
		ESCRIBIR(ARCH_SAL,REG_SAL);
		LEER(ARCH_PEL,REG_P);
	FM;
